#include <iostream>

using namespace std;

int main()
{
    int a,b,kol=0;
    int begin=1;
 while (begin)   {
    cout<<"Введите а"<< endl;
    cin >>a;
    cout<<"Введите b"<< endl;
    cin >>b;

    for (int i = a; i<=b; i++ ) {
        if (i%2 != 0) {
            kol++;
            }
        }
    if (a >= b) {
    cout<<"Ошибка... Повторно введите число a = ";
    cin>>a;
    }
    cout<<"Количество нечетных чисел =  "<< kol << endl;
    cout<<"Повторить? (1/0)"<< endl;
    cin>>begin;
    if(begin == 0)
    {
        exit(0);
    }
 }
}
