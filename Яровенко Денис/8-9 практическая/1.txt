#include <iostream>
using namespace std;
float volume (float a, float b, float c)
{
    
      return a*b*c; 
}


int main()
{
    float a,b,c,v;
    cout<<"Введите a,b,c"<< endl;
    cin>>a>>b>>c;
    if (a <= 0 || b <= 0 || c <= 0) {
        cout<<"Вы ввели не верное число ";
        exit(0);
    }
    v= volume(a,b,c);
    cout<<"V="<<v;
    return 0;
}
